# Packages installed on every host.
#
# apt-get install -y is a no-op when a package is already at the
# pinned version, so this list is safe to re-run. Apps that need
# anything more specialized (wireguard, app-specific runtimes, …)
# install it in their own init steps.

# === Sysadmin baseline ===

# Source control + downloads
git
curl
wget

# Editors + process tools
vim
htop
unzip
xxd

# Archiving
p7zip-full

# DNS troubleshooting (dig, nslookup, host)
dnsutils

# Network troubleshooting (ss, sockstat, traceroute, tcpdump)
tcpdump
traceroute
iproute2

# Trust anchors (apt + curl)
ca-certificates

# Privilege escalation — needed so subsequent steps can hand off
# to the deploy user.
sudo

# === Dev/ops ===

# Apt repository signing keyrings (used by 50-docker + any app that
# adds 3rd-party apt repos).
gnupg

# Template rendering — `envsubst` lives in gettext-base. Needed by
# any app's render step that processes config templates.
gettext-base

# JSON / crypto helpers used by init.sh health checks and secret
# generation across the monorepo.
jq
openssl

# Per-directory environment loading (.envrc → shell env)
direnv

# Native Node module compilation (node-gyp)
build-essential

# === Developer tooling ===

# Shell script linting
shellcheck

# Fast recursive code search
ripgrep

# Terminal multiplexer for persistent remote sessions
tmux

# Process / port diagnostics
lsof

# File transfer / sync between hosts
rsync

# Tab completion in interactive shells
bash-completion
